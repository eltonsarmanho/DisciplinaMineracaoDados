\subsection{Testes de Hipótese}

% --- Frame: O que é Teste de Hipótese ---
\begin{frame}{O que é um Teste de Hipótese?}
    \begin{block}{Definição}
        \footnotesize
        Procedimento para \textbf{decidir}, com base em evidências amostrais, se uma afirmação sobre uma população deve ser aceita ou rejeitada.
    \end{block}
    \vspace{-0.3cm}
    \pause
    \begin{columns}
        \column{0.7\textwidth}
        \begin{block}{Componentes Essenciais}
            \begin{itemize}\footnotesize
                \item \textbf{$H_0$ (hipótese nula):} posição padrão\\
                      ex.\ ``RF e SVM têm mesma acurácia''
                \item \textbf{$H_1$ (alternativa):} o que queremos provar
                \item \textbf{Nível de significância $\alpha$:} geralmente $0{.}05$
                \begin{itemize}
                    \item Probabilidade de rejeitar $H_0$ quando $H_0$ é verdadeira (Erro Tipo I)
                \end{itemize}
                \item \textbf{p-valor:} rejeita $H_0$ se $p < \alpha$
                \begin{itemize}
                    \item Probabilidade de observar os dados assumindo $H_0$ verdadeira
                \end{itemize}
            \end{itemize}
        \end{block}
        \pause
        \column{0.45\textwidth}
        \centering
        \begin{tikzpicture}[scale=0.72,font=\tiny,
            box/.style={rectangle,draw=mineblue,fill=mineblue!10,
                        rounded corners,minimum width=2.8cm,
                        minimum height=0.65cm,align=center},
            arr/.style={-{Stealth[length=5pt]},mineblue,thick}]
            \node[box] (h)   at (0,0)    {Formular $H_0$, $H_1$};
            \node[box] (d)   at (0,-1.0) {Coletar Dados};
            \node[box] (t)   at (0,-2.0) {Calcular estatística};
            \node[box] (p)   at (0,-3.0) {Calcular p-valor};
            \node[box,fill=minegreen!15,draw=minegreen]
                       (dec) at (0,-4.0) {Decisão: $p < \alpha$?};
            \node[font=\tiny,minered]  (rej) at ( 1.8,-4.9) {Rejeitar $H_0$};
            \node[font=\tiny,minegreen](nrj) at (-1.8,-4.9) {Não rejeitar $H_0$};
            \draw[arr] (h)--(d); \draw[arr] (d)--(t);
            \draw[arr] (t)--(p); \draw[arr] (p)--(dec);
            \draw[arr,minered]   (dec.south east) -- (rej.north);
            \draw[arr,minegreen] (dec.south west) -- (nrj.north);
        \end{tikzpicture}
    \end{columns}
\end{frame}

% --- Frame: Tipos de Erro ---
\begin{frame}{Tipos de Erro e Poder do Teste}
    \begin{block}{Tabela de Decisão}
        \begin{center}\footnotesize
        \begin{tabular}{lcc}
        \toprule
        & $H_0$ verdadeira & $H_0$ falsa \\
        \midrule
        Não rejeitar & \textcolor{minegreen}{\faCheck\ Correto} & \textcolor{minered}{Erro Tipo II ($\beta$)} \\
        Rejeitar     & \textcolor{minered}{Erro Tipo I ($\alpha$)} & \textcolor{minegreen}{\faCheck\ Poder ($1-\beta$)} \\
        \bottomrule
        \end{tabular}
        \end{center}
    \end{block}
    \pause
    \begin{alertblock}{Erros}
        \footnotesize
        \begin{itemize}
            \item \textbf{Erro Tipo I ($\alpha$):} rejeitar $H_0$ quando $H_0$ é verdadeira
            \item \textbf{Erro Tipo II ($\beta$):} não rejeitar $H_0$ quando $H_0$ é falsa
            \item \textbf{Poder do teste ($1-\beta$):} probabilidade de rejeitar $H_0$ quando $H_0$ é falsa
        \end{itemize}
    \end{alertblock}
    
\end{frame}

\begin{frame}{Tipos de Erro e Poder do Teste}
    
    \begin{block}{Testes Relevantes em Mineração de Dados}
        \begin{itemize}\footnotesize
            \item \textbf{Teste $t$ pareado:} compara 2 classificadores nos mesmos folds
            \item \textbf{ANOVA:} compara 3 ou mais classificadores simultaneamente
            \item \textbf{Wilcoxon Signed-Rank:} alternativa não-paramétrica ao teste $t$
            \item \textbf{McNemar:} compara classificadores pelas matrizes de confusão
        \end{itemize}
    \end{block}
\end{frame}

% --- Frame: Teste t Pareado --- Fórmula e quando usar ---
\begin{frame}{Teste $t$ Pareado: Fórmula e Quando Usar}
    \begin{columns}
        \column{0.5\textwidth}
        \begin{block}{Fórmula}
            \[
              t = \frac{\bar{d}}{s_d / \sqrt{n}}, \quad
              \bar{d} = \frac{1}{n}\sum_{i=1}^{n}(a_i - b_i)
            \]
            \footnotesize $\bar{d}$: média das diferenças por fold; $s_d$: desvio padrão.
        \end{block}
        \pause
        \begin{block}{Quando Usar}
            \begin{itemize}\footnotesize
                \item Dois algoritmos nos \textit{mesmos} folds (amostras pareadas)
                \item Diferenças aproximadamente normais
                \item Para $n < 30$, verificar normalidade das diferenças
            \end{itemize}
        \end{block}
        \column{0.5\textwidth}
        \pause
        \begin{block}{Passos}
            \begin{enumerate}\footnotesize
                \item Formular $H_0$, $H_1$ e escolher $\alpha$
                \item Calcular $d_i = a_i - b_i$ para cada fold
                \item Calcular $\bar{d}$, $s_d$ e a estatística $t$
                \item Comparar com $t_{\text{crítico}}(gl=n-1,\;\alpha)$
            \end{enumerate}
        \end{block}
 
    \end{columns}
\end{frame}

% --- FRAME 1: O Determinante ---
\begin{frame}{O que determina o tipo de teste?}
    \begin{block}{A Hipótese Alternativa ($H_1$)}
        A escolha é definida \textbf{antes} da coleta de dados, baseada na pergunta de pesquisa:
    \end{block}
    
    \begin{itemize}
        \item \textbf{Teste Bilateral (Two-tailed):} Quando queremos detectar qualquer diferença, sem direção preferencial.
        \begin{itemize}
            \item $H_1: \mu \neq \mu_0$
        \end{itemize}
        \vspace{0.3cm}
        \item \textbf{Teste Unilateral (One-tailed):} Quando a teoria indica que o efeito só pode (ou só interessa) ocorrer em um sentido.
        \begin{itemize}
            \item $H_1: \mu > \mu_0$ (Direita) ou $H_1: \mu < \mu_0$ (Esquerda)
        \end{itemize}
    \end{itemize}
\end{frame}

% --- FRAME 2: Diferenças Fundamentais ---
\begin{frame}{Diferenças: Bilateral vs. Unilateral}
    \begin{table}[]
        \small
        \begin{tabular}{|l|l|l|}
        \hline
        \textbf{Característica} & \textbf{Bilateral} & \textbf{Unilateral} \\ \hline
        \textbf{Direção} & Indiferente ($\neq$) & Específica ($>$ ou $<$) \\ \hline
        \textbf{Distribuição $\alpha$} & Dividido ($\alpha/2$ cada lado) & Total em uma cauda \\ \hline
        \textbf{Rigor} & Mais conservador & Menos conservador \\ \hline
        \textbf{Poder} & Menor poder estatístico & Maior poder (na direção certa) \\ \hline
        \end{tabular}
    \end{table}
    
    \begin{alertblock}{Importante}
        No teste bilateral, o valor crítico de $t$ ou $z$ é sempre \textbf{maior}, pois a área de rejeição é empurrada para os extremos.
    \end{alertblock}
\end{frame}

    \begin{block}{Cenário}
        \footnotesize
        \textbf{Random Forest} vs \textbf{SVM} em detecção de spam --- 10 folds de validação cruzada.
    \end{block}
    \pause
    \begin{columns}
        \column{0.42\textwidth}
        \begin{center}\footnotesize
        \begin{tabular}{ccc}
        \toprule
        Fold & RF (\%) & SVM (\%) \\
        \midrule
        1  & 92{.}1 & 89{.}3 \\
        2  & 88{.}7 & 86{.}0 \\
        3  & 91{.}4 & 90{.}1 \\
        4  & 93{.}0 & 88{.}5 \\
        5  & 89{.}5 & 87{.}9 \\
        6  & 94{.}2 & 91{.}6 \\
        7  & 90{.}3 & 89{.}0 \\
        8  & 88{.}0 & 85{.}5 \\
        9  & 92{.}8 & 90{.}4 \\
        10 & 91{.}0 & 88{.}7 \\
        \bottomrule
        \end{tabular}
        \end{center}
        \column{0.58\textwidth}
        \pause
        \begin{block}{Hipóteses (unilateral)}
            \footnotesize
            \begin{itemize}
                \item $H_0$: $\mu_{\text{RF}} = \mu_{\text{SVM}}$
                \item $H_1$: $\mu_{\text{RF}} > \mu_{\text{SVM}}$ (RF é melhor)
                \item $\alpha = 0{.}05$
            \end{itemize}
        \end{block}
        \pause
        \begin{block}{Teste Escolhido}
            \footnotesize
            Mesmos folds $\Rightarrow$ \textbf{Teste $t$ pareado} (amostras dependentes).
        \end{block}
    \end{columns}
\end{frame}

% --- Frame: Exemplo 1 — Cálculo de t ---
\begin{frame}[shrink=10]{Exemplo 1 - RF vs SVM: Cálculo de $t$}
    \begin{center}\footnotesize
    \begin{tabular}{cccc}
    \toprule
    Fold & RF & SVM & $d_i = \text{RF}-\text{SVM}$ \\
    \midrule
    1  & 92{.}1 & 89{.}3 & $+2{.}8$ \\
    2  & 88{.}7 & 86{.}0 & $+2{.}7$ \\
    3  & 91{.}4 & 90{.}1 & $+1{.}3$ \\
    4  & 93{.}0 & 88{.}5 & $+4{.}5$ \\
    5  & 89{.}5 & 87{.}9 & $+1{.}6$ \\
    6  & 94{.}2 & 91{.}6 & $+2{.}6$ \\
    7  & 90{.}3 & 89{.}0 & $+1{.}3$ \\
    8  & 88{.}0 & 85{.}5 & $+2{.}5$ \\
    9  & 92{.}8 & 90{.}4 & $+2{.}4$ \\
    10 & 91{.}0 & 88{.}7 & $+2{.}3$ \\
    \midrule
    \multicolumn{3}{r}{\textbf{Média $\bar{d}$}} & $2{.}40$ \\
    \multicolumn{3}{r}{\textbf{Desvio padrão $s_d$}} & $0{.}91$ \\
    \bottomrule
    \end{tabular}
    \end{center}
    \pause
    \[
      t = \frac{\bar{d}}{s_d/\sqrt{n}} = \frac{2{.}40}{0{.}91/\sqrt{10}} = \frac{2{.}40}{0{.}288} \approx \mathbf{8{.}33}
    \]
    \pause
    \begin{alertblock}{Decisão}
        \footnotesize $t_{\text{crítico}}(gl=9,\;\alpha=0{.}05) = 1{.}833$.\quad Como $8{.}33 \gg 1{.}833$ e $p < 0{.}001 (computado)$ $\rightarrow$ \textbf{rejeitamos $H_0$} $(p <\alpha)$.\\
        RF é estatisticamente superior ao SVM. $\Delta = 2{.}4\%$, logo avalie relevância prática no domínio.
    \end{alertblock}
\end{frame}

\begin{frame}[shrink=8]{Exemplo 2 — MLP vs Reg.\ Log.: Baixo Poder com $n=5$}
    \begin{block}{Cenário}
        \footnotesize MLP vs Regressão Logística, 5 folds, diagnóstico médico.\\
        $H_0: \mu_d = 0$ \; vs \; $H_1: \mu_d \neq 0$ \; (bilateral, $\alpha=0{.}05$)
    \end{block}
    \pause
    \begin{columns}
        \column{0.42\textwidth}
        \begin{center}\footnotesize
        \begin{tabular}{cccc}
        \toprule
        Fold & MLP & Reg.\ Log. & $d_i$ \\
        \midrule
        1 & 85{.}0 & 80{.}0 & $+5{.}0$ \\
        2 & 90{.}0 & 85{.}0 & $+5{.}0$ \\
        3 & 78{.}0 & 80{.}0 & $-2{.}0$ \\
        4 & 92{.}0 & 87{.}0 & $+5{.}0$ \\
        5 & 88{.}0 & 85{.}0 & $+3{.}0$ \\
        \midrule
        & $\bar{d}$ & & $3{.}20$ \\
        & $s_d$     & & $2{.}77$ \\
        \bottomrule
        \end{tabular}
        \end{center}
        \column{0.58\textwidth}
        \pause
        \[
          t = \frac{3{.}20}{2{.}77/\sqrt{5}} \approx 2{.}58
        \]
        \pause
        \begin{alertblock}{Resultado}
            \footnotesize $t_{\text{crit}}(gl=4,\;\alpha=0{.}025)=2{.}776$.\\
            $2{.}58 < 2{.}776$: \textbf{não rejeitamos $H_0$} $(p\approx0{.}06)$.
        \end{alertblock}
        \pause
        \begin{block}{Lição: Poder do Teste}
            \footnotesize Com apenas $n=5$ folds, o teste tem \textbf{poder reduzido}. O resultado é \textit{borderline}, repetir com 10+ folds pode revelar diferença significativa.
        \end{block}
    \end{columns}
\end{frame}

% --- Frame: Exemplo 3 — ANOVA — Cálculo F ---
\begin{frame}[shrink=5]{Exemplo 3 — ANOVA: Cálculo da Estatística $F$}
    \footnotesize
    \[
      SS_{\text{entre}} = n\sum_{j}(\bar{x}_j-\bar{x})^2
        = 5\bigl[(83{-}79{.}27)^2 + (79{.}8{-}79{.}27)^2 + (75{-}79{.}27)^2\bigr] = 162{.}1
    \]
    \pause
    \[
      SS_{\text{dentro}} = \sum_j\sum_i(x_{ij}-\bar{x}_j)^2
        = \underbrace{10}_{\text{KNN}} + \underbrace{8{.}8}_{\text{DT}} + \underbrace{10}_{\text{NB}} = 28{.}8
    \]
    \pause
    \begin{center}\footnotesize
    \begin{tabular}{lcccc}
    \toprule
    Fonte & $SS$ & $gl$ & $MS$ & $F$ \\
    \midrule
    Entre grupos & $162{.}1$ & $2$ & $81{.}05$ & $\mathbf{33{.}77}$ \\
    Dentro grupos & $28{.}8$  & $12$ & $2{.}40$ & \\
    Total & $190{.}9$ & $14$ & & \\
    \bottomrule
    \end{tabular}
    \end{center}
    \pause
    \begin{alertblock}{Decisão e Post-hoc}
        \footnotesize $F_{\text{crit}}(2,12) = 3{.}89$. Como $F = 33{.}77 \gg 3{.}89$: \textbf{rejeitamos $H_0$} $(p < 0{.}001)$.\\
        ANOVA indica diferença, mas \textbf{não diz qual par} difere. Use \textbf{Tukey HSD} ou \textbf{Bonferroni}.\\
        KNN ($83\%$) supera NB ($75\%$) por 8 p.p.\ Diferença estatística \textit{e} prática.
    \end{alertblock}
\end{frame}

% --- Frame: Pearson — Definição ---
\begin{frame}{Correlação de Pearson: Definição}
    \begin{block}{Fórmula}
        \[
          r = \frac{\sum_{i=1}^{n}(x_i-\bar{x})(y_i-\bar{y})}
                   {\sqrt{\sum_{i=1}^{n}(x_i-\bar{x})^2 \cdot \sum_{i=1}^{n}(y_i-\bar{y})^2}}
          \in [-1,\;1]
        \]
    \end{block}
    \pause
    \begin{columns}
        \column{0.5\textwidth}
        \begin{block}{Escala de Interpretação}
            \begin{center}\footnotesize
            \begin{tabular}{cc}
            \toprule
            $|r|$ & Interpretação \\
            \midrule
            $0{.}90$--$1{.}00$ & Muito forte \\
            $0{.}70$--$0{.}89$ & Forte \\
            $0{.}50$--$0{.}69$ & Moderada \\
            $0{.}30$--$0{.}49$ & Fraca \\
            $< 0{.}30$        & Desprezível \\
            \bottomrule
            \end{tabular}
            \end{center}
        \end{block}
        \column{0.5\textwidth}
        \pause
        \begin{alertblock}{Limitações}
            \footnotesize
            \begin{itemize}
                \item Mede apenas relação \textbf{linear}
                \item Sensível a outliers
                \item \textbf{Correlação $\neq$ Causalidade}
                \item Para relações não-lineares: use Spearman ou Informação Mútua
            \end{itemize}
        \end{alertblock}
    \end{columns}
\end{frame}

% --- Frame: Pearson — Exemplo com dados ---
\begin{frame}[shrink=10]{Exemplo — Pearson: Horas de Estudo vs Nota Final}
    \begin{block}{Cenário}
        \footnotesize 8 alunos. Existe correlação linear entre horas de estudo semanal e nota final (0--10)?
    \end{block}
    \pause
    \begin{center}\tiny
    \begin{tabular}{cccccc}
    \toprule
    Aluno & $x_i$ (h) & $y_i$ (nota) & $(x_i-\bar{x})$ & $(y_i-\bar{y})$ & $(x_i-\bar{x})(y_i-\bar{y})$ \\
    \midrule

% --- Frame: Spearman, Kendall, IM ---
\begin{frame}[shrink=5]{Spearman, Kendall e Informação Mútua}
    \begin{columns}
        \column{0.5\textwidth}
        \begin{block}{Correlação de Spearman ($\rho$)}
            \footnotesize
            Aplica Pearson sobre os \textbf{postos} (\textit{ranks}) dos dados.
            \begin{itemize}\footnotesize
                \item Detecta relações \textbf{monotônicas} (não só lineares)
                \item Robusta a outliers
                \item Indicada para dados ordinais ou distribuição assimétrica
            \end{itemize}
        \end{block}
        \pause
        \begin{block}{Tau de Kendall ($\tau$)}
            \footnotesize
            Baseado em pares \textbf{concordantes/discordantes}.
            \begin{itemize}\footnotesize
                \item Mais robusto que Spearman para $n$ pequeno
                \item Interpretação probabilística direta
            \end{itemize}
        \end{block}
        \column{0.5\textwidth}
        \pause
        \begin{block}{Informação Mútua $I(X;Y)$}
            \[
              I(X;Y) = \sum_{x,y} p(x,y)\log\frac{p(x,y)}{p(x)\,p(y)}
            \]
            \begin{itemize}\footnotesize
                \item Mede dependência \textbf{geral} (não apenas linear)
                \item $I(X;Y) = 0$ somente se $X$ e $Y$ são independentes
                \item Usado em seleção de features: \textbf{MRMR}
                \item Captura relações complexas que Pearson ignora
            \end{itemize}
        \end{block}
        \pause
        \begin{alertblock}{}
            \footnotesize Prefira Spearman ou IM quando há suspeita de relação não-linear ou presença de outliers.
        \end{alertblock}
    \end{columns}
\end{frame}

% --- Frame: K-Means conceito ---
\begin{frame}[shrink=7]{K-Means: Conceito e Objetivo}
    \begin{columns}
        \column{0.5\textwidth}
        \begin{block}{Objetivo: Minimizar Inércia}
            \[
              J = \sum_{k=1}^{K} \sum_{\mathbf{x}_i \in C_k}
                  \|\mathbf{x}_i - \boldsymbol{\mu}_k\|^2
            \]
            onde $\boldsymbol{\mu}_k$ é o centróide do cluster $k$.
        \end{block}
        \begin{block}{Resumo do Algoritmo (Lloyd, 1982)}
            \begin{enumerate}\footnotesize
                \item Inicializar $K$ centróides (aleatório ou K-Means++)
                \item \textbf{Atribuição:} cada ponto vai ao centróide mais próximo
                \item \textbf{Atualização:} mover centróide para a média do cluster
                \item Repetir 2--3 até convergência
            \end{enumerate}
        \end{block}
        \column{0.5\textwidth}
        \begin{block}{Propriedades}
            \begin{itemize}\footnotesize
                \item \textbf{Complexidade:} $O(n \cdot K \cdot d \cdot i)$, onde $d$ é a dimensão e $i$ o número de iterações
                \item \textbf{Convergência:} garantida (mas pode ser ótimo local)
                \item \textbf{Limitações:} assume clusters \textbf{esféricos e de tamanho similar}; sensível a outliers e inicialização
                \item \textbf{Distância:} euclidiana por padrão
            \end{itemize}
        \end{block}
        \vspace{0.1cm}
        \centering\tiny \faArrowRight\ Consulte Material Didático e Colab para implementação em Python
    \end{columns}
\end{frame}

% --- Frame: K-Means passo a passo ---

% --- Frame: Silhouette ---
\begin{frame}[shrink=5]{Escolha de K: Análise de Silhouette}
    \begin{columns}
        \column{0.50\textwidth}
        \begin{block}{Coeficiente de Silhouette}
            \footnotesize
            Para cada ponto $\mathbf{x}_i$ no cluster $C_k$:
            \begin{align*}
              a(i) &= \text{distância média intra-cluster}\\
              b(i) &= \text{distância média ao vizinho mais próximo}
            \end{align*}
            \[
              s(i) = \frac{b(i)-a(i)}{\max\{a(i),b(i)\}} \in [-1,\,1]
            \]
        \end{block}
        \begin{block}{Interpretação}
            \begin{itemize}\footnotesize
                \item $s \approx +1$: ponto bem dentro do cluster
                \item $s \approx 0$: ponto na fronteira
                \item $s \approx -1$: ponto no cluster errado
                \item $\bar{s} > 0{.}5$: clustering de qualidade
            \end{itemize}
        \end{block}
        \column{0.50\textwidth}
        \centering
        \begin{tikzpicture}[scale=0.85]
        \begin{axis}[

% --- Frame: Variantes K-Means ---
\begin{frame}[shrink=5]{Variantes do K-Means}
    \begin{columns}
        \column{0.5\textwidth}
        \begin{block}{K-Means++}
            \footnotesize
            Inicialização inteligente: cada centróide é escolhido com probabilidade proporcional à distância ao centróide mais próximo já escolhido.
            \begin{itemize}\footnotesize
                \item Evita inicializações ruins
                \item Garante aproximação $O(\log K)$ do ótimo
                \item \textbf{Padrão} no scikit-learn
            \end{itemize}
        \end{block}
        \begin{block}{K-Medoids (PAM)}
            \footnotesize
            Usa exemplos reais como centróides (medóides).
            \begin{itemize}\footnotesize
                \item Mais robusto a outliers que K-Means
                \item Interpretável: centróide é um ponto real
                \item Mais custoso computacionalmente
            \end{itemize}
        \end{block}
        \column{0.5\textwidth}
        \begin{block}{Mini-Batch K-Means}
            \footnotesize
            Atualiza centróides com mini-lotes aleatórios a cada iteração.
            \begin{itemize}\footnotesize
                \item Escala para datasets de \textbf{milhões} de exemplos
                \item Resultado ligeiramente pior que K-Means padrão
                \item Ideal para dados em stream
            \end{itemize}
        \end{block}
        \begin{alertblock}{Quando K-Means falha}
            \footnotesize Clusters não esféricos, de tamanhos muito diferentes, ou com muitos outliers. Nesses casos, prefira DBSCAN, GMM ou Spectral Clustering.
        \end{alertblock}
    \end{columns}
\end{frame}

\subsection{Clustering Hierárquico}
% ==============================================================================

% --- Frame: Clustering Hierárquico ---
\begin{frame}[shrink=12]{Clustering Hierárquico}
    \begin{columns}
        \column{0.50\textwidth}
        \begin{block}{Conceito}
            \footnotesize
            Cria uma \textbf{hierarquia} de agrupamentos em um \textbf{dendrograma}, sem necessitar especificar $K$ antecipadamente.
        \end{block}
        \begin{block}{Abordagens}
            \begin{itemize}\footnotesize
                \item \textbf{Agglomerativo (bottom-up):} começa com $N$ clusters e mescla os dois mais próximos iterativamente
                \item \textbf{Divisivo (top-down):} começa com um cluster e divide recursivamente
            \end{itemize}
        \end{block}
        \vspace{-0.25cm}
        \begin{block}{Critérios de Ligação}
            \begin{itemize}\footnotesize
                \item \textbf{Single:} distância mínima, sensível a outliers
                \item \textbf{Complete:} distância máxima, clusters compactos
                \item \textbf{Average (UPGMA):} equilíbrio entre os dois
                \item \textbf{Ward:} minimiza variância intra-cluster (+popular)

\begin{frame}[shrink=5]{DBSCAN: Clustering Baseado em Densidade}

\begin{frame}{HDBSCAN: Extensão Hierárquica do DBSCAN}

        \column{0.5\textwidth}
        \begin{block}{Limitação do DBSCAN}
            \footnotesize

            \end{itemize}
        \end{block}

% --- Frame: Métricas de Avaliação ---
\begin{frame}[shrink=5]{Avaliação de Agrupamentos}
    \begin{columns}
        \column{0.5\textwidth}
        \begin{block}{Métricas Internas (sem rótulos)}
            \footnotesize
            \begin{itemize}
                \item \textbf{Silhouette Score $\bar{s}$:} coesão interna vs separação externa. $\bar{s} \in [-1,1]$, maximize.
                \item \textbf{Davies-Bouldin (DB):}
                      $\text{DB} = \frac{1}{K}\sum_{k}\max_{k'\neq k}\frac{s_k+s_{k'}}{\|c_k-c_{k'}\|}$\\
                      Minimize. Penaliza clusters compactos mas muito próximos.
                \item \textbf{Calinski-Harabasz:} razão dispersão inter/intra-cluster. Maximize.
            \end{itemize}
        \end{block}
        \column{0.5\textwidth}
        \begin{block}{Métricas Externas (com rótulos verdadeiros)}
            \footnotesize
            \begin{itemize}
                \item \textbf{ARI (Adjusted Rand Index):} $\in [-1,1]$; 1 = perfeito; 0 = aleatório. Ajustado para chance.
                \item \textbf{NMI (Normalized Mutual Information):}
                      $\text{NMI}(U,V) = \frac{2\cdot I(U;V)}{H(U)+H(V)} \in [0,1]$
                \item \textbf{V-Measure:} média harmônica de \textit{homogeneidade} e \textit{completude}.
            \end{itemize}
        \end{block}
    \end{columns}
    \vspace{0.2cm}
    \begin{alertblock}{Boas Práticas}
        \centering\footnotesize Nunca use apenas uma métrica. Combine métricas \textbf{internas} + \textbf{externas} (quando disponíveis) + visualização (PCA, t-SNE, UMAP).
    \end{alertblock}
\end{frame}

\subsection{Conceitos Fundamentais}

% --- Frame: Medidas de Interesse (Explicação) ---



\begin{frame}{Medidas de Interesse em Regras de Associação}
    \begin{block}{O que é uma Regra? $A \Rightarrow B$}
        \footnotesize
        Uma implicação entre dois itemsets: \textbf{se o cliente compra A, qual é a probabilidade de também comprar B?}
    \end{block}
    \pause
    \begin{columns}
        \column{0.5\textwidth}
        \begin{block}{\textbf{Suporte} (frequência absoluta)}
            \footnotesize
            $\text{sup}(A \Rightarrow B) = \frac{|\{T: A \cup B \subseteq T\}|}{|\mathcal{T}|}$\\[4pt]
            Proporção de transações contendo \textbf{tanto A quanto B}.\\[4pt]
            \textbf{Filtro:} elimina regras raras; regras úteis têm sup $> 1\%$ tipicamente.
        \end{block}      
        \column{0.5\textwidth}
        \pause
        \begin{block}{\textbf{Lift} (correlação)}
            \footnotesize
            $\text{lift}(A \Rightarrow B) = \frac{\text{conf}(A \Rightarrow B)}{\text{sup}(B)}$\\[4pt]
            Quanto a presença de A \textbf{amplifica} a probabilidade de B.\\[4pt]
            \begin{itemize}\footnotesize
                \item lift $= 1$: A e B são independentes
                \item lift $> 1$: \textit{associação positiva} (correlação favorável)
                \item lift $< 1$: \textit{associação negativa} (correlação desfavorável)
            \end{itemize}
        \end{block}
       
    \end{columns}
\end{frame}

\begin{frame}{Medidas de Interesse em Regras de Associação}
    \begin{block}{O que é uma Regra? $A \Rightarrow B$}
        \footnotesize
        Uma implicação entre dois itemsets: \textbf{se o cliente compra A, qual é a probabilidade de também comprar B?}
    \end{block}
    \pause
    \begin{columns}
        \column{0.5\textwidth}
        
        \begin{block}{\textbf{Confiança} (probabilidade condicional)}
            \footnotesize
            $\text{conf}(A \Rightarrow B) = \frac{\text{sup}(A \cup B)}{\text{sup}(A)}$\\[4pt]
            \textbf{P(B $|$ A)} de quem compra A, quantos também compram B.\\[4pt]
            Intervalo: [0, 1]. Regras úteis têm conf $> 50\%$ tipicamente.
        \end{block}
        \column{0.5\textwidth}
        
        \begin{block}{\textbf{Convicção}}
            \footnotesize
            $\text{conv}(A \Rightarrow B) = \frac{1 - \text{sup}(B)}{1 - \text{conf}(A \Rightarrow B)}$\\[4pt]
            Força da implicação: quanto a regra contradiz a independência.\\[4pt]
            Conv $\to \infty$: relação quase determinística.
        \end{block}
    \end{columns}
\end{frame}
% --- Frame: Medidas de Interesse ---
\begin{frame}[shrink=5]{Regras de Associação: Conceitos}
    \begin{columns}
        \column{0.50\textwidth}
        \begin{block}{O Problema}
            \footnotesize
            Dado um banco de transações, encontrar regras do tipo $A \Rightarrow B$ com alta frequência e confiabilidade.\\[4pt]
            \textbf{Exemplo:} $\{\text{leite}\} \Rightarrow \{\text{pão}\}$
        \end{block}
        \begin{block}{Medidas de Interesse}
            \footnotesize
            \begin{align*}
              \text{sup}(A{\Rightarrow}B) &= \tfrac{|\{T: A\cup B\subseteq T\}|}{|\mathcal{T}|}\\
              \text{conf}(A{\Rightarrow}B) &= \tfrac{\text{sup}(A\cup B)}{\text{sup}(A)}\\
              \text{lift}(A{\Rightarrow}B) &= \tfrac{\text{conf}(A\Rightarrow B)}{\text{sup}(B)}
            \end{align*}
        \end{block}
        \column{0.50\textwidth}
        \begin{block}{Interpretação do Lift}
            \begin{itemize}\footnotesize
                \item \textbf{lift $> 1$:} correlação positiva; a regra é mais frequente que o esperado por acaso
                \item \textbf{lift $= 1$:} $A$ e $B$ independentes
                \item \textbf{lift $< 1$:} correlação negativa (anti-regra)
            \end{itemize}
        \end{block}
        \begin{block}{Exemplo: Cesta de Mercado}
            \footnotesize
            5 transações. Para $\{\text{leite}\} \Rightarrow \{\text{pão}\}$ vamos verificar as associações
            \faCheck\ Será que há correlação positiva: comprar leite favorece comprar pão ?
        \end{block}
    \end{columns}
\end{frame}

% --- Frame: Exemplo Detalhado ---
\begin{frame}[shrink=3]{Exemplo Detalhado: Cesta de Mercado}
    \begin{block}{Base de Transações (5 transações)}
        \begin{center}\footnotesize
        \begin{tabular}{ccccc}
        \toprule
        $T_1$ & $T_2$ & $T_3$ & $T_4$ & $T_5$ \\
        \midrule
        \textcolor{mineblue}{\textbf{Leite}} & \textcolor{mineblue}{\textbf{Leite}} & Manteiga & \textcolor{mineblue}{\textbf{Leite}} & Queijo \\
        \textcolor{mineorange}{\textbf{Pão}} & Manteiga & \textcolor{mineorange}{\textbf{Pão}} & \textcolor{mineorange}{\textbf{Pão}} & \\
        & & Queijo & & \\
        \bottomrule
        \end{tabular}
        \end{center}
    \end{block}
    \pause
    \begin{columns}
        \column{0.5\textwidth}
        \begin{block}{Contagens}
            \footnotesize
            \begin{itemize}
                \item Transações com \textbf{Leite} $(A)$: $T_1, T_2, T_4$ → 3 transações
                \item Transações com \textbf{Pão} $(B)$: $T_1, T_3, T_4$ → 3 transações
                \item Transações com \textbf{Leite E Pão} $(A \cup B)$: $T_1, T_4$ → 2 transações
                \item Total: 5 transações
            \end{itemize}
        \end{block}
        \pause
        \begin{block}{Cálculo de Suporte}
            \footnotesize
            $\text{sup}(A \Rightarrow B) = \frac{|\{T: \text{Leite} \cap \text{Pão}\}|}{|\mathcal{T}|}$\\[6pt]
            $= \frac{2}{5} = \boxed{0{.}40}$\\[6pt]
            \textit{40\% das transações contêm leite \textbf{e} pão}
        \end{block}
        \column{0.5\textwidth}
        \pause
        \begin{block}{Cálculo de Confiança}
            \footnotesize
            $\text{conf}(A \Rightarrow B) = \frac{\text{sup}(A \cap B)}{\text{sup}(A)}$\\[6pt]
            $= \frac{2/5}{3/5} = \frac{2}{3} = \boxed{0{.}67}$\\[6pt]
            \textit{De quem compra leite, 67\% também compram pão}
        \end{block}
        \pause
        \begin{block}{Cálculo de Lift}
            \footnotesize
            $\text{lift}(A \Rightarrow B) = \frac{\text{conf}(A \Rightarrow B)}{\text{sup}(B)}$\\[6pt]
            $= \frac{0{.}67}{3/5} = \frac{2/3}{0{.}60} = \boxed{1{.}11}$\\[6pt]
            \textit{Comprar leite aumenta a probabilidade de pão em 11\%}
        \end{block}
    \end{columns}
    \pause
    \begin{alertblock}{Interpretação Integrada}
        \footnotesize
        A regra $\{\text{Leite}\} \Rightarrow \{\text{Pão}\}$ tem \textbf{suporte baixo} (40\%, menos de 50\%), \textbf{confiança moderada} (67\%) e \textbf{lift positivo} (1.11).
        A conclusão: \textbf{leite e pão são positivamente correlacionados}, mas a regra não é frequente o bastante para ser considerada de alto suporte em um cenário real.
    \end{alertblock}
\end{frame}

\begin{frame}[shrink=6]{Tipos de Anomalias}

            \begin{itemize}\footnotesize
                \item \textbf{Ponto:} instância individual muito diferente das demais\\

% --- Frame: Métodos de Detecção ---
\begin{frame}[shrink=5]{Métodos de Detecção de Anomalias}
    \begin{columns}
        \column{0.5\textwidth}
        \begin{block}{Métodos Estatísticos}
            \begin{itemize}\footnotesize
                \item \textbf{Z-score:} anomalia se $|z| > 3$. Assume distribuição Gaussiana.
                \item \textbf{Boxplot:} outlier se $> Q_3 + 1{.}5 \cdot \text{IQR}$ ou $< Q_1 - 1{.}5 \cdot \text{IQR}$
                \item \textbf{Grubbs Test:} teste formal para outlier único em amostra normal
            \end{itemize}
        \end{block}
        \begin{block}{LOF (Breunig et al., 2000)}
            \footnotesize
            Compara a \textbf{densidade local} de um ponto com a de seus vizinhos. $\text{LOF}(p) \gg 1$ indica anomalia.
            \begin{itemize}\footnotesize
                \item Detecta outliers locais (não depende de distribuição global)
                \item Sensível ao parâmetro $k$ (número de vizinhos)
            \end{itemize}
        \end{block}
        \column{0.5\textwidth}
        \begin{block}{Isolation Forest (Liu et al., 2008)}
            \footnotesize
            Constrói árvores de isolamento aleatórias. Anomalias são isoladas em \textbf{caminhos mais curtos}.
            \[
              s(x,n) = 2^{-E[h(x)]/c(n)}
            \]
            \begin{itemize}\footnotesize
                \item Eficiente em alta dimensão
                \item Não assume forma da distribuição
                \item Eficaz para anomalias de ponto
            \end{itemize}
        \end{block}
        \begin{block}{Autoencoder}
            \footnotesize
            Treina para reconstruir dados normais. \textbf{Alto erro de reconstrução} $\|x-\hat{x}\|^2$ indica anomalia.
        \end{block}
        \centering\tiny \faArrowRight\ Consulte Colab para comparação prática de métodos
    \end{columns}
\end{frame}

% --- Frame: Deep Clustering ---
\begin{frame}[shrink=5]{Estado da Arte: Deep Clustering}
    \begin{columns}
        \column{0.5\textwidth}
        \begin{block}{DEC (Xie et al., 2016)}
            \footnotesize
            \textbf{Deep Embedded Clustering:} autoencoder treinado conjuntamente com objetivo de clustering. Aprende representações otimizadas para agrupamento.
        \end{block}
        \begin{block}{SCAN (Van Gansbeke et al., 2020)}
            \footnotesize
            \textbf{Self-supervised clustering:} usa similaridade de vizinhos para treinar sem rótulos. Estado da arte em clustering de imagens.
        \end{block}
        \begin{block}{DINO (Caron et al., 2021)}
            \footnotesize
            Vision Transformer (ViT) auto-supervisionado revela estrutura de clustering sem fine-tuning. Propriedade emergente dos Transformers.
        \end{block}
        \column{0.5\textwidth}
        \begin{block}{Detecção de Outliers Moderna}
            \begin{itemize}\footnotesize
                \item \textbf{PyOD} (Zhao et al., 2019): 40+ algoritmos unificados em API sklearn-compatível
                \item \textbf{COPOD} (Li et al., 2020): baseado em cópulas, sem hiperparâmetros, muito eficiente
                \item \textbf{ECOD} (Li et al., 2022): baseado em distribuição empírica cumulativa, interpretável
                \item \textbf{Anomalia em grafos:} GraphSAGE + Isolation Forest para fraudes em redes sociais
            \end{itemize}
        \end{block}
        \vspace{0.1cm}
        \centering\tiny \faArrowRight\ Material Didático e Colab trazem experimentos com BERTopic e PyOD
    \end{columns}
\end{frame}

% --- Frame: BERTopic e Pipeline moderno ---
\begin{frame}{Pipeline Moderno: BERTopic e UMAP+HDBSCAN}
    \begin{columns}
        \column{0.52\textwidth}
        \begin{block}{BERTopic (Grootendorst, 2022)}
            \footnotesize
            Modelagem de tópicos combinando:
            \begin{enumerate}\footnotesize
                \item \textbf{BERT embeddings:} representação semântica dos textos
                \item \textbf{UMAP:} redução de dimensionalidade
                \item \textbf{HDBSCAN:} agrupamento dos embeddings
                \item \textbf{TF-IDF por cluster:} extração de palavras-chave representativas
            \end{enumerate}
            Supera LDA em qualidade de tópicos.
        \end{block}
        \begin{alertblock}{}
            \footnotesize Pipeline UMAP + HDBSCAN tornou-se padrão para NLP, genômica single-cell (scRNA-seq) e análise de documentos.
        \end{alertblock}
