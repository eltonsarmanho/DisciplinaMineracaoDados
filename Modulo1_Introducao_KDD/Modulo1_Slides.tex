\section{O que é Mineração de Dados}
% ==============================================================================

\begin{frame}{Definição Formal}
    \begin{block}{Definição Clássica (Fayyad et al., 1996)}
        Mineração de Dados é a extração de informação \textbf{implícita},
        \textbf{previamente desconhecida} e \textbf{potencialmente útil}
        a partir de dados armazenados em grandes bases de dados.
    \end{block}

    \vspace{0.4cm}
    \begin{columns}[T]
        \begin{column}{0.32\textwidth}
            \begin{block}{\faSearch\ Implícita}
                Não diretamente observável — requer inferência e busca.
            \end{block}
        \end{column}
        \begin{column}{0.32\textwidth}
            \begin{block}{\faLightbulbO\ Desconhecida}
                Novo conhecimento — não confirma hipóteses já sabidas.
            \end{block}
        \end{column}
        \begin{column}{0.32\textwidth}
            \begin{block}{\faCogs\ Útil}
                Valor prático ou teórico — passível de ação.
            \end{block}
        \end{column}
    \end{columns}

    \vspace{0.3cm}
    \begin{center}
        \small\textit{Para detalhes, consulte o \textbf{Material Didático -- Módulo 1}.}
    \end{center}
\end{frame}

\begin{frame}{Exemplos Práticos}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}

\section{Processo KDD}
% ==============================================================================

\begin{frame}{Etapas do Processo KDD (Fayyad, 1996)}
    \begin{columns}[T]
        \begin{column}{0.38\textwidth}
            \begin{center}
            \begin{tikzpicture}[
                stepbox/.style={rectangle,rounded corners,fill=mineblue!20,draw=mineblue,
                             minimum width=2.6cm,minimum height=0.65cm,align=center,font=\scriptsize},
                arrow/.style={->,thick,mineblue},
                scale=0.9]
            \node[stepbox] (s1) at (0,0) {1. Seleção de Dados};
            \node[stepbox] (s2) at (0,-1.0) {2. Pré-processamento};
            \node[stepbox] (s3) at (0,-2.0) {3. Transformação};
            \node[stepbox] (s4) at (0,-3.0) {4. Mineração de Dados};
            \node[stepbox,fill=minegreen!20,draw=minegreen] (s5) at (0,-4.0) {5. Avaliação};
            \node[stepbox,fill=mineorange!20,draw=mineorange] (s6) at (0,-5.0) {6. Utilização};

            \draw[arrow] (s1)--(s2);
            \draw[arrow] (s2)--(s3);
            \draw[arrow] (s3)--(s4);
            \draw[arrow] (s4)--(s5);
            \draw[arrow] (s5)--(s6);

            \draw[arrow,dashed,mineorange] (s5.east) -- ++(0.8,0) -- ++(0,2.0) -- (s3.east);
            \draw[arrow,dashed,minered] (s4.east) -- ++(0.5,0) -- ++(0,1.0) -- (s2.east);
            \end{tikzpicture}
            \end{center}
        \end{column}
        \begin{column}{0.58\textwidth}
            \begin{itemize}
                \item[\textbf{1.}] \textbf{Seleção:} definir universo e fontes de dados
                \item[\textbf{2.}] \textbf{Pré-processamento:} tratar valores ausentes, outliers e inconsistências \\ {\tiny(60--80\% do esforço!)}
                \item[\textbf{3.}] \textbf{Transformação:} normalização, encoding, feature engineering
                \item[\textbf{4.}] \textbf{Mineração:} aplicar algoritmos para descobrir padrões
                \item[\textbf{5.}] \textbf{Avaliação:} validar resultados com métricas e domínio
                \item[\textbf{6.}] \textbf{Utilização:} integrar o conhecimento a sistemas reais
            \end{itemize}
            \vspace{0.3cm}
            \begin{alertblock}{}
                \tiny Processo \textbf{iterativo e interativo} — feedbacks podem reiniciar etapas anteriores.
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}

        \begin{column}{0.46\textwidth}
            \vspace{0.3cm}
            \textbf{CRISP-DM} (\textit{Cross-Industry Standard Process for Data Mining}) é o padrão de facto na indústria.
            \vspace{0.4cm}

            \begin{itemize}\small
                \item Foco no \textbf{negócio} desde o início
                \item Processo \textbf{cíclico} e iterativo
                \item Adotado por empresas e academia
            \end{itemize}
            \vspace{0.4cm}
            \begin{block}{}
                \tiny \textbf{Diferença para o KDD:} O CRISP-DM incorpora explicitamente a perspectiva de negócios e é mais orientado à prática industrial.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\section{Ética e LGPD}
% ==============================================================================

\begin{frame}{Ética e Responsabilidade em Mineração de Dados}
    \begin{alertblock}{\faWarning\ Atenção: Ética não é Opcional}
        Vieses em dados de treino podem causar discriminação sistemática. Ignorar a LGPD pode resultar em multas de até 2\% do faturamento (limite de R\$~50 milhões por infração).
    \end{alertblock}

    \vspace{0.3cm}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \begin{block}{\faBalanceScale\ Viés Algorítmico}
                Modelos treinados com dados históricos podem perpetuar desigualdades (ex.: caso COMPAS na Justiça americana).
            \end{block}
            \vspace{0.2cm}
            \begin{block}{\faLock\ Privacidade de Dados}
                Dados pessoais sensíveis exigem anonimização eficaz, consentimento informado e minimização de uso.
            \end{block}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{block}{\faGavel\ LGPD (Lei 13.709/2018)}
                Direitos dos titulares: acesso, correção, exclusão e portabilidade. Todo projeto com dados pessoais deve estar em conformidade.
            \end{block}
            \vspace{0.2cm}
            \begin{block}{\faEye\ Transparência Algorítmica}
                O ``direito de explicação'' impõe restrições ao uso de modelos caixa-preta em contextos de alto impacto.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}
